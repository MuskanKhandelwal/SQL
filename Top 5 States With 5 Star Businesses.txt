WITH state_counts AS (
    SELECT
        state,
        COUNT(*) AS n_businesses
    FROM yelp_business
    WHERE stars = 5
    GROUP BY state
)
SELECT
    state,
    n_businesses
FROM (
    SELECT
        *,
        DENSE_RANK() OVER (ORDER BY n_businesses DESC) AS rnk
    FROM state_counts
) ranked
WHERE rnk <= 5
ORDER BY
    n_businesses DESC,
    state ASC;
